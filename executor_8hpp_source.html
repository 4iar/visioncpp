110 /// \struct SubExprExecute
111 /// \brief it is used to statically determine whether or not a subexpression
112 /// execution is needed. It increases the execution time by avoiding executing
113 /// the subexpression when it is not needed. Using this
114 /// struct with sub_expression_evaluation parameter in every non-terminal node,
115 /// it is possible to determine such condition at compile time.
116 template <bool Val, bool ExecPolicy, size_t LC, size_t LR, size_t LCT,
117  size_t LRT, typename Expr>
118 struct SubExprExecute {
119  template <typename DeviceT>
120  static void inline execute(Expr &expr, const DeviceT &dev) {
121  expr.reset(true);
122  Executor<ExecPolicy, LC, LR, LCT, LRT,
123  decltype(expr.template sub_expression_evaluation<false, LC, LR,
124  LCT, LRT>(dev))>::
125  execute(
126  expr.template sub_expression_evaluation<false, LC, LR, LCT, LRT>(
127  dev),
128  dev);
129  }
130 };
