29 /// \brief the FuseExpr when the expression type is not a terminal node
30 /// (leafNode).
31 template <size_t LCIn, size_t LRIn, size_t LCT, size_t LRT, typename Expr>
32 struct FuseExpr {
33  /// \brief the fuse function for executing the given expr.
34  /// \param expr : the expression passed to be executed on the device
35  /// \param dev : the selected device for executing the expression
36  /// return void
37  template <typename DeviceT>
38  static void fuse(Expr &expr, const DeviceT &dev) {
39  /// LRT is the workgroup size row and is checked with LR. LR is based
40  /// on LRIn. LCT is the workgroup size column checked with LC. LC is based
41  /// on LCIn. The local memory size and the work group size is calculated at
42  /// compile time.
43  constexpr size_t LR =
44  tools::IfConst<(Expr::RThread > LRIn), LRIn, Expr::RThread>::Value;
45  constexpr size_t LC =
46  tools::IfConst<(Expr::CThread > LCIn), LCIn, Expr::CThread>::Value;
47 
48  constexpr int rLThread =
49  tools::IfConst<(Expr::RThread > LRT), LRT, Expr::RThread>::Value;
50 
51  constexpr int cLThread =
52  tools::IfConst<(Expr::CThread > LCT), LCT, Expr::CThread>::Value;
53 
54  constexpr size_t rGThreads =
55  (tools::IfConst<(Expr::RThread % LR == 0), (Expr::RThread / LR),
56  ((Expr::RThread / LR) + 1)>::Value) *
57  rLThread;
58 
59  constexpr size_t cGThreads =
60  (tools::IfConst<(Expr::CThread % LC == 0), (Expr::CThread / LC),
61  ((Expr::CThread / LC) + 1)>::Value) *
62  cLThread;
63  dev.template execute<LC, LR, cGThreads, rGThreads, cLThread, rLThread>(
64  expr);
65  }
66 };
